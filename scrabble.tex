\newcommand{\scrabbletile}[3]{

        \node[draw,thick,fill=quitelightgrey,minimum size=0.9cm,rounded corners=0.1cm] at (0,0) {};
        \node at (-0.05,0) {\Large #2};
        \node at (0.35,-0.25) {\footnotesize #3};
    \end{scope}
}
